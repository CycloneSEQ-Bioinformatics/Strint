outdir <- tempfile() #创建一个临时目录
dir.create(outdir) #创建一个临时目录
# some example data
# known cell barcodes, e.g. from coupled short-read sequencing
bc_allow <- file.path(outdir, "bc_allow.tsv") #设置一个bc_allow.tsv文件
R.utils::gunzip(
  filename = system.file("extdata", "bc_allow.tsv.gz", package = "FLAMES"),
  destname = bc_allow, remove = FALSE
) #将镜像中的文件解压到指定文件
# reference genome
genome_fa <- file.path(outdir, "rps24.fa") #设置一个参考fasta
R.utils::gunzip(
  filename = system.file("extdata", "rps24.fa.gz", package = "FLAMES"),
  destname = genome_fa, remove = FALSE
) #将指定文件解压到fasta

#构建一个pipeline
pipeline <- SingleCellPipeline(
  # use the default configs
  #设置config相关信息
  config_file = create_config(
    outdir,
    pipeline_parameters.demultiplexer = "flexiplex" #利用flexiplex进行barcode拆分
  ),
  outdir = outdir,
  # the input fastq file
  fastq = system.file("extdata", "fastq", "musc_rps24.fastq.gz", package = "FLAMES"),
  # reference annotation file
  annotation = system.file("extdata", "rps24.gtf.gz", package = "FLAMES"),
  genome_fa = genome_fa,
  barcodes_file = bc_allow
)
